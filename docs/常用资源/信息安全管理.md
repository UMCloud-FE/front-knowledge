```bash
# 1、修改nginx配置
vim /etc/nginx/nginx.conf
log_format main '$remote_addr - "$http_user_agent"'
access_log /root/access.log

# 2、提取异常日志

cat /root/access.log | awk '$3' | grep MSJ/1.0 > /root/danger.log
```
